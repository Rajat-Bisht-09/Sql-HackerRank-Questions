SELECT DISTINCT(CITY)
FROM STATION WHERE 
(LOWER(CITY) LIKE 'a%' or LOWER(CITY) LIKE 'e%' OR LOWER(CITY) LIKE 'i%'  OR LOWER(CITY) LIKE 'o%' OR LOWER(CITY) LIKE 'u%') 
AND
(LOWER(CITY) LIKE '%a' or LOWER(CITY) LIKE '%e' OR LOWER(CITY) LIKE '%i'  OR
LOWER(CITY) LIKE '%o' OR LOWER(CITY) LIKE '%u')  

OR

SELECT DISTINCT(CITY)
FROM STATION 
WHERE LOWER(CITY) REGEXP '^[aeiou].*[aeiou]$'
